// ============================================================
// DeepSync - Firmware ESP32 GPRS (SIM800L)
// ============================================================

#include <Wire.h>
#include <Adafruit_ADXL345_U.h>
#include <HX711.h>
#include <ArduinoJson.h>
#include <HardwareSerial.h>

// ============================================================
// CONFIGURATION
// ============================================================
const char* DEVICE_ID = "NODE_01";
const char* PROJECT_ID = "PROJ_001";
const char* SERVER_HOST = "votre-serveur.com";
const int SERVER_PORT = 8000;

// ============================================================
// BROCHES
// ============================================================
#define HX711_DOUT 2
#define HX711_SCK 3
#define ADXL_SDA 21
#define ADXL_SCL 22
#define SIM_RX 27
#define SIM_TX 26
#define BUZZER_PIN 13
#define LED_RED 14
#define LED_GREEN 15

HardwareSerial sim800(1);

// ============================================================
// OBJETS
// ============================================================
Adafruit_ADXL345_Unified accel = Adafruit_ADXL345_Unified(12345);
HX711 scale;

// ============================================================
// VARIABLES
// ============================================================
float s1_mm = 0.0;
float sh_mm = 0.0;
float piezo_kPa = 0.0;
float ratio_ds = 0.0;
float s_curve = 0.0;
unsigned long lastRead = 0;

// ============================================================
// SETUP
// ============================================================
void setup() {
  Serial.begin(115200);
  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(LED_RED, OUTPUT);
  pinMode(LED_GREEN, OUTPUT);
  
  // HX711
  scale.begin(HX711_DOUT, HX711_SCK);
  scale.set_scale(2.205);
  scale.tare();
  
  // ADXL345
  Wire.begin(ADXL_SDA, ADXL_SCL);
  if (!accel.begin()) {
    Serial.println("[ERROR] ADXL345 not found!");
  }
  accel.setRange(ADXL345_RANGE_2_G);
  
  // SIM800L
  sim800.begin(9600, SERIAL_8N1, SIM_RX, SIM_TX);
  delay(3000);
  simInit();
  
  digitalWrite(LED_GREEN, HIGH);
  Serial.println("[INIT] DeepSync GPRS ready");
}

// ============================================================
// LOOP
// ============================================================
void loop() {
  unsigned long now = millis();
  
  if (now - lastRead >= 30000) {
    lastRead = now;
    readAllSensors();
    calculateStability();
    checkAlarms();
    
    // Envoi via GPRS
    sendHTTPPost(s1_mm, sh_mm, piezo_kPa);
  }
}

// ============================================================
// FONCTIONS
// ============================================================
void readAllSensors() {
  float raw = scale.get_units(10);
  s1_mm = raw;
  if (s1_mm < 0) s1_mm = 0;
  
  sensors_event_t event;
  accel.getEvent(&event);
  float angle_rad = atan2(event.acceleration.x, event.acceleration.z);
  float hauteur_remblai_m = 5.0;
  sh_mm = sin(angle_rad) * hauteur_remblai_m * 1000.0;
  if (sh_mm < 0) sh_mm = -sh_mm;
  
  int piezo_raw = analogRead(34);
  piezo_kPa = (piezo_raw / 4095.0) * 500.0;
  
  Serial.printf("[SENSOR] S1=%.2fmm | Sh=%.2fmm | Piezo=%.1fkPa\n",
                s1_mm, sh_mm, piezo_kPa);
}

void calculateStability() {
  if (s1_mm > 0.1) {
    ratio_ds = sh_mm / s1_mm;
  } else {
    ratio_ds = 0.0;
  }
  s_curve = 5.93 * exp(1.28 * ratio_ds * ratio_ds - 3.41 * ratio_ds);
  Serial.printf("[CALC] δ/s=%.3f | s_courbe=%.2f\n", ratio_ds, s_curve);
}

void checkAlarms() {
  bool danger = (s1_mm > s_curve);
  if (danger) {
    digitalWrite(LED_RED, HIGH);
    digitalWrite(LED_GREEN, LOW);
    for (int i = 0; i < 3; i++) {
      tone(BUZZER_PIN, 1000, 200);
      delay(300);
    }
    sendSMSAlarm("+213XXXXXXXXX", "ALARME DANGER! S=" + String(s1_mm) + "mm");
  } else {
    digitalWrite(LED_RED, LOW);
    digitalWrite(LED_GREEN, HIGH);
    noTone(BUZZER_PIN);
  }
}

// ============================================================
// SIM800L / GPRS
// ============================================================
bool simInit() {
  sim800.println("AT");
  delay(1000);
  sim800.println("AT+CPIN?");
  delay(1000);
  sim800.println("AT+CREG?");
  delay(1000);
  sim800.println("AT+SAPBR=3,1,\"Contype\",\"GPRS\"");
  delay(500);
  sim800.println("AT+SAPBR=3,1,\"APN\",\"internet\"");
  delay(500);
  sim800.println("AT+SAPBR=1,1");
  delay(3000);
  sim800.println("AT+SAPBR=2,1");
  delay(500);
  Serial.println("[SIM800L] GPRS connected");
  return true;
}

void sendHTTPPost(float s1, float sh, float piezo) {
  String payload = "{\"device_id\":\"" + String(DEVICE_ID) + "\",";
  payload += "\"project_id\":\"" + String(PROJECT_ID) + "\",";
  payload += "\"s1_mm\":" + String(s1) + ",";
  payload += "\"sh_mm\":" + String(sh) + ",";
  payload += "\"piezo_kPa\":" + String(piezo) + "}";
  
  sim800.println("AT+HTTPINIT");
  delay(500);
  sim800.println("AT+HTTPPARA=\"CID\",1");
  delay(500);
  sim800.println("AT+HTTPPARA=\"URL\",\"http://" + String(SERVER_HOST) + ":" + String(SERVER_PORT) + "/api/mqtt/publish\"");
  delay(500);
  sim800.println("AT+HTTPPARA=\"CONTENT\",\"application/json\"");
  delay(500);
  sim800.println("AT+HTTPDATA=" + String(payload.length()) + ",10000");
  delay(500);
  sim800.print(payload);
  delay(2000);
  sim800.println("AT+HTTPACTION=1");
  delay(3000);
  sim800.println("AT+HTTPTERM");
  Serial.println("[GPRS] Data sent");
}

void sendSMSAlarm(String phone, String message) {
  sim800.println("AT+CMGF=1");
  delay(500);
  sim800.println("AT+CMGS=\"" + phone + "\"");
  delay(500);
  sim800.print(message);
  sim800.write(26);
  delay(3000);
  Serial.println("[SMS] Sent to " + phone);
}
